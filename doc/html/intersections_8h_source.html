44 bool intersectionRayNSphere(Ray<dim> ray, NSphere<dim> nsphere, double &time, Point<dim>& point) {
45  Vector<dim> f = ray.o - nsphere.c;
46 
47  double a = ray.d * ray.d;
48  double b = (f * ray.d) * 2;
49  double c = (f * f) - (nsphere.r * nsphere.r);
50 
51  if (c > 0 && b > 0) return (false);
52 
53  double discr = b * b - 4 * a * c;
54 
55  if (discr < 0) return (false);
56 
57  double t = (-b - sqrt(discr)) / (2 * a);
58  if (t < 0) t = 0;
59  if (t > time) return (false);
60  time = t;
61  point = ray.o + ray.d * time;
62 
63  return (true);
64 }

91  for(unsigned i = 0; i < dim; i++) {

98

100

102  double t1 = n1 * ood;
103  double t2 = n2 * ood;
104 
105  if(t1 > t2) {std::swap(t1,t2);}
106  if(t1 > tmin) {tmin = t1;}
107  if(t2 < tmax) {tmax = t2;}

109  if(tmin > tmax) {return (0);}
110  } // for each dimension
111 
112  if(tmin > time) {return (false);}
113 
114  time = tmin;
115  point = ray.o + (ray.d * time);
116  return (true);
